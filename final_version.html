		SELECT *, EXTRACT (YEAR FROM "publishedDate") AS year
		FROM books_data
		WHERE "ratingsCount" IS NOT NULL AND "categories" IS NOT NULL
	) AS sub 
	GROUP BY year, categories
	ORDER BY year, books_count DESC;
    """

import matplotlib.pyplot as plt
import pandas as pd
df['publishedDate'] = pd.to_datetime(df['publishedDate'], errors='coerce')
df['year'] = df['publishedDate'].dt.year
scatter_data = df[['year', 'ratingsCount']].dropna()
scatter_data = scatter_data[(scatter_data['year'] >= 1900) & (scatter_data['year'] <= 2025)]
plt.figure(figsize=(10, 6))
plt.scatter(scatter_data['year'], scatter_data['ratingsCount'], alpha=0.5)
plt.title('Publication Year vs. Number of Ratings')
plt.xlabel('Year')
plt.ylabel('Number of Ratings')
plt.grid(True)
plt.yscale('log') 
plt.tight_layout()
plt.show()
